ວິທີການສ້າງຜູ້ໃຊ້ດ້ວຍເຄື່ອງມືນໍາເຂົ້າ

ບົດຄວາມນີ້ສະແດງໃຫ້ເຫັນຜູ້ເບິ່ງແຍງວິທີການສ້າງຜູ້ໃຊ້ດ້ວຍເຄື່ອງມືການນໍາເຂົ້າ.

ເປີດ​ເຄື່ອງ​ມື​ການ​ນໍາ​ເຂົ້າ​

ຜູ້ເບິ່ງແຍງສາມາດໃຊ້ເຄື່ອງມືການນໍາເຂົ້າເພື່ອອັບໂຫລດຜູ້ໃຊ້ໃຫມ່ຈໍານວນຫລາຍແລະດັດແປງບັນທຶກຜູ້ໃຊ້ທີ່ມີຢູ່.

  1. ເຂົ້າສູ່ລະບົບບັນຊີ admin.
  2. ຊອກຫາແອັບຯໃນຫນ້າ Apps ແລະຄລິກໃສ່ປຸ່ມ ADMIN.
  3. ໃນກະດານ Admin, ໃຫ້ຄລິກໃສ່ແຖບການລວມເຂົ້າກັນແລະຫຼັງຈາກນັ້ນ IMPORTS ໃນເມນູຍ່ອຍ.
  4. ໃຫ້ຄລິກໃສ່ປຸ່ມ NEW. ກ່ອງໂຕ້ຕອບ Add Imports ສະແດງຢູ່ໃນແຖບຕົວທ່ອງເວັບໃຫມ່.

ການສ້າງບັນທຶກ

ຫຼັງຈາກກ່ອງໂຕ້ຕອບ Add Imports ເປີດ:

  1. ໃຫ້ຄລິກໃສ່ ປະເພດ ພາກສະຫນາມແລະເລືອກທາງເລືອກ (ເບິ່ງຂ້າງລຸ່ມນີ້). ການເຊື່ອມຕໍ່ສະແດງ.
  2. ຄລິກທີ່ລິ້ງ ດາວໂຫຼດຕົວຢ່າງ . ໄຟລ໌ CSV ດາວໂຫຼດໃສ່ອຸປະກອນຂອງທ່ານ.

ຕໍ່ໄປ, ເປີດໄຟລ໌ແລະເພີ່ມບັນທຶກ. ຄລິກ ທີ່ນີ້ ເພື່ອຮຽນຮູ້ວິທີໃຊ້ຂໍ້ມູນຮູບແບບໃນໄຟລ໌ CSV.

ປະເພດ

ເມື່ອ​ທ່ານ​ຄລິກ​ໃສ່​ພາກ​ສະ​ຫນາມ​ປະ​ເພດ​, ເປັນ​ເມ​ນູ​ເລື່ອນ​ລົງ​ສະ​ແດງ​. ເລືອກຫນຶ່ງໃນສອງທາງເລືອກສໍາລັບການນໍາເຂົ້າຈໍານວນຫຼາຍ.

:exclamation: ຫມາຍເຫດ ເມນູເລື່ອນລົງມີຫຼາຍທາງເລືອກ, ແຕ່ມີພຽງສອງອັນທີ່ກ່ຽວຂ້ອງກັບເຄື່ອງມືນໍາເຂົ້າ. ແຕ່ລະທາງເລືອກມີໄຟລ໌ CSV ຕົວຢ່າງທີ່ແຕກຕ່າງກັນ.

ເພີ່ມຜູ້ໃຊ້

  • ເລືອກຕົວເລືອກນີ້ສໍາລັບການອັບໂຫລດບັນທຶກຜູ້ໃຊ້ຫຼາຍອັນງ່າຍດາຍ.
  • ຄລິກທີ່ລິ້ງດາວໂຫຼດຕົວຢ່າງເພື່ອເປີດໄຟລ໌ CSV ທີ່ມີຫົວຂໍ້ຖັນສຳລັບ Add users Import type.

ແກ້ໄຂຜູ້ໃຊ້

  • ເລືອກຕົວເລືອກນີ້ເພື່ອອັບເດດຂໍ້ມູນສຳລັບຜູ້ໃຊ້ປັດຈຸບັນ.
  • ຄລິກທີ່ລິ້ງດາວໂຫຼດຕົວຢ່າງເພື່ອເປີດໄຟລ໌ CSV ທີ່ມີຫົວຂໍ້ຖັນສຳລັບ Modify users Import type.

ກຳລັງອັບໂຫລດໄຟລ໌

ທ່ານສາມາດອັບໂຫລດໄຟລ໌ໄດ້ທັນທີ ຫຼືກັບຄືນໄປທີ່ກ່ອງໂຕ້ຕອບເພີ່ມການນໍາເຂົ້າໃນພາຍຫຼັງ ແລະເຮັດສໍາເລັດຂັ້ນຕອນເຫຼົ່ານີ້.

  1. ເປີດກ່ອງໂຕ້ຕອບເພີ່ມການນໍາເຂົ້າ.
  2. ໃສ່ຊື່ສະເພາະສໍາລັບໄຟລ໌ CSV.
  3. ຄລິກໄອຄອນບວກ (+) ເພື່ອອັບໂຫລດໄຟລ໌ຈາກອຸປະກອນຂອງທ່ານ.
  4. ໃຫ້ຄລິກໃສ່ ປະເພດ ພາກສະຫນາມແລະເລືອກເອົາທາງເລືອກ. (ເບິ່ງ​ຢູ່​ຂ້າງ​ເທິງ)
  5. ກົດ SAVE.

ການປຸງແຕ່ງ

ອີງຕາມຈໍານວນບັນທຶກໃນໄຟລ໌ແລະການໂຫຼດຂອງລະບົບ, ເວທີສາມາດໃຊ້ເວລາຫຼາຍນາທີເພື່ອປະມວນຜົນການນໍາເຂົ້າ.

ເມື່ອການນໍາເຂົ້າສໍາເລັດ, ເວທີຈະປັບບັນທຶກຜູ້ໃຊ້ (ເຊັ່ນ: ເພີ່ມ, ແກ້ໄຂ, ຫຼືປິດການໃຊ້ງານ) ໃນ dashboard ຜູ້ໃຊ້ຂອງຜູ້ ເບິ່ງແຍງ.

ຄວາມລົ້ມເຫຼວ

ຖ້າການປະຕິບັດການນໍາເຂົ້າລົ້ມເຫລວ, ເວທີຈະສົ່ງຄືນຂໍ້ຄວາມທີ່ລົ້ມເຫລວ. ໃນກໍລະນີຫຼາຍທີ່ສຸດ, ສາເຫດຂອງຄວາມລົ້ມເຫຼວແມ່ນການຈັດຮູບແບບຂໍ້ມູນ.

ເພື່ອເບິ່ງສາເຫດຂອງຄວາມລົ້ມເຫລວ:

  1. ກົດ ເບິ່ງ ພາຍໃຕ້ຫົວຂໍ້ການປະຕິບັດ.
  2. ໃນໜ້າຕໍ່ໄປ, ເລື່ອນເມົ້າໄປທີ່ໄອຄອນທີ່ລົ້ມເຫລວ. ຄໍາແນະນໍາເຄື່ອງມືສະແດງຂໍ້ຜິດພາດ.
  3. ໄປທີ່ໄຟລ໌ CVS ແລະປັບແລະແກ້ໄຂຂໍ້ຜິດພາດ.
  4. ອັບໂຫຼດໄຟລ໌ອີກຄັ້ງ.
